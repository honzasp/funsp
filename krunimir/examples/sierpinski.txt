define sierpinski(n) {
  if(n) {
    sierpinski(n-1)
    left(45) forward(9) left(45)
    sierpinski(n-1)
    right(90) forward(7) right(90)
    sierpinski(n-1)
    left(45) forward(9) left(45)
    sierpinski(n-1)
  }
  if(1-n) {
    forward(7)
  }
}
