define dragonA(n,step) {
  if(n) {
    right(45)
    dragonA(n-1,step)
    left(90)
    dragonB(n-1,step)
    right(45)
  }
  if(1-n) {
    forward(step)
  }
}

define dragonB(n,step) {
  if(n) {
    left(45)
    dragonA(n-1,step)
    right(90)
    dragonB(n-1,step)
    left(45)
  }
  if(1-n) {
    forward(step)
  }
}
